Binary Search


def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1

Bubble Sort Pseudocode


ALGORITHM BubbleSort(A)
    for i = 0 to A.length - 1
        for j = 0 to A.length - i - 1
            if A[j] > A[j + 1]
                swap A[j] and A[j + 1]
    return A
